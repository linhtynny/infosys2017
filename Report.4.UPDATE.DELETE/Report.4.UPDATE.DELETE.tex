\section{Insert new departments}
\begin{lstlisting}[language=sql]
insert into departments values
('d010', 'Research and Development'), 
('d011', 'Marketing and Sales');
\end{lstlisting}
\includegraphics[width=\linewidth]{1.png}

\section{Move employees from 'Development' and 'Research' to new department 'Research and Development'}
\begin{lstlisting}[language=sql]
update ignore dept_emp
set dept_no = 'd010'
where dept_no = 'd005' or dept_no = 'd008';
\end{lstlisting}
\includegraphics[width=\linewidth]{2.png}

\section{Move managers from 'Development' and 'Research' to new department 'Research and Development'}
\begin{lstlisting}[language=sql]

\end{lstlisting}

\section{Delete old departments}
\begin{lstlisting}[language=sql]
delete from departments
where dept_no = 'd005' or dept_no = 'd008' 
or dept_no = 'd001' or dept_no = 'd007';
\end{lstlisting}
